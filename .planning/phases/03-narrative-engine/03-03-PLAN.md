```go
package database

import (
    "context"
    "database/sql"
    "encoding/json"
    "fmt"
    "time"
)

type Narrative struct {
    ID              int64  `json:"id"`
    IncidentID      int64  `json:"incident_id"`
    Summary         string `json:"summary"`
    Confidence      float64 `json:"confidence"`
    Sentences       json.RawMessage `json:"sentences"`
    ModelUsed       string `json:"model_used"`
    Temperature     float64 `json:"temperature"`
    TokensUsed      int    `json:"tokens_used"`
    GenerationTimeMs int   `json:"generation_time_ms"`
    CreatedAt       string `json:"created_at"`
    UpdatedAt       string `json:"updated_at"`
}

type NarrativeSentence struct {
    Text           string  `json:"text"`
    Timestamp      string  `json:"timestamp"`
    SourceEventIDs []int64 `json:"source_event_ids"`
    Confidence     float64 `json:"confidence"`
    Technique      string  `json:"technique,omitempty"`
}

func (db *DB) SaveNarrative(n *Narrative) error {
    sentencesJSON, err := json.Marshal(n.Sentences)
    if err != nil {
        return fmt.Errorf("marshal sentences: %w", err)
    }
    
    query := `
        INSERT INTO narratives (incident_id, summary, confidence, sentences, model_used, temperature, tokens_used, generation_time_ms)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    `
    
    result, err := db.conn.Exec(query,
        n.IncidentID, n.Summary, n.Confidence, sentencesJSON,
        n.ModelUsed, n.Temperature, n.TokensUsed, n.GenerationTimeMs,
    )
    if err != nil {
        return fmt.Errorf("insert narrative: %w", err)
    }
    
    id, err := result.LastInsertId()
    if err != nil {
        return fmt.Errorf("get last insert id: %w", err)
    }
    n.ID = id
    return nil
}

func (db *DB) GetNarrativeByIncident(incidentID int64) (*Narrative, error) {
    query := `
        SELECT id, incident_id, summary, confidence, sentences, model_used, temperature, tokens_used, generation_time_ms, created_at, updated_at
        FROM narratives
        WHERE incident_id = ?
        ORDER BY created_at DESC
        LIMIT 1
    `
    
    n := &Narrative{}
    var sentencesJSON []byte
    err := db.conn.QueryRow(query, incidentID).Scan(
        &n.ID, &n.IncidentID, &n.Summary, &n.Confidence, &sentencesJSON,
        &n.ModelUsed, &n.Temperature, &n.TokensUsed, &n.GenerationTimeMs,
        &n.CreatedAt, &n.UpdatedAt,
    )
    if err == sql.ErrNoRows {
        return nil, nil
    }
    if err != nil {
        return nil, fmt.Errorf("query narrative: %w", err)
    }
    
    n.Sentences = json.RawMessage(sentencesJSON)
    return n, nil
}

func (db *DB) GetNarrativeByID(id int64) (*Narrative, error) {
    query := `
        SELECT id, incident_id, summary, confidence, sentences, model_used, temperature, tokens_used, generation_time_ms, created_at, updated_at
        FROM narratives
        WHERE id = ?
    `
    
    n := &Narrative{}
    var sentencesJSON []byte
    err := db.conn.QueryRow(query, id).Scan(
        &n.ID, &n.IncidentID, &n.Summary, &n.Confidence, &sentencesJSON,
        &n.ModelUsed, &n.Temperature, &n.TokensUsed, &n.GenerationTimeMs,
        &n.CreatedAt, &n.UpdatedAt,
    )
    if err == sql.ErrNoRows {
        return nil, nil
    }
    if err != nil {
        return nil, fmt.Errorf("query narrative: %w", err)
    }
    
    n.Sentences = json.RawMessage(sentencesJSON)
    return n, nil
}

func (db *DB) EventExists(id int64) (bool, error) {
    var count int
    err := db.conn.QueryRow("SELECT COUNT(*) FROM events WHERE id = ?", id).Scan(&count)
    return count > 0, err
}
```

**Step 2: Create Narrative Handler (internal/handler/narrative_handler.go)**

Create new handler for narrative API endpoints:

```go
package handler

import (
    "context"
    "net/http"
    "strconv"
    "time"
    
    "github.com/gin-gonic/gin"
    "github.com/superduperpiyuxh/narrator-ai/backend/internal/database"
    "github.com/superduperpiyuxh/narrator-ai/backend/internal/llm"
    "github.com/superduperpiyuxh/narrator-ai/backend/internal/narrative"
)

type NarrativeHandler struct {
    db        *database.DB
    llmClient *llm.Client
}

func NewNarrativeHandler(db *database.DB, llmClient *llm.Client) *NarrativeHandler {
    return &NarrativeHandler{db: db, llmClient: llmClient}
}

func (h *NarrativeHandler) GenerateNarrative(c *gin.Context) {
    incidentID, err := strconv.ParseInt(c.Param("id"), 10, 64)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "invalid incident id"})
        return
    }
    
    // Check if incident exists
    incident, err := h.db.GetIncidentByID(incidentID)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }
    if incident == nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "incident not found"})
        return
    }
    
    // Check if narrative already exists
    existing, err := h.db.GetNarrativeByIncident(incidentID)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }
    if existing != nil {
        c.JSON(http.StatusOK, gin.H{"narrative": existing, "cached": true})
        return
    }
    
    // Get events for this incident
    events, err := h.db.GetIncidentEvents(incidentID)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }
    if len(events) == 0 {
        c.JSON(http.StatusBadRequest, gin.H{"error": "no events found for incident"})
        return
    }
    
    // Generate narrative
    startTime := time.Now()
    ctx, cancel := context.WithTimeout(c.Request.Context(), 60*time.Second)
    defer cancel()
    
    narrResponse, err := narrative.GenerateNarrative(ctx, h.llmClient, incident, events)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "narrative generation failed: " + err.Error()})
        return
    }
    generationTime := time.Since(startTime).Milliseconds()
    
    // Validate output (per SECU-03)
    if err := narrative.ValidateNarrativeOutput(h.db, narrResponse); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "narrative validation failed: " + err.Error()})
        return
    }
    
    // Calculate confidence score
    confidence := narrative.CalculateConfidence(incident, events, narrResponse)
    narrResponse.Confidence = confidence
    
    // Store narrative
    sentencesJSON, _ := json.Marshal(narrResponse.Sentences)
    dbNarrative := &database.Narrative{
        IncidentID:       incidentID,
        Summary:          narrResponse.Summary,
        Confidence:       confidence,
        Sentences:        sentencesJSON,
        ModelUsed:        h.llmClient.Model(),
        Temperature:      0.2,
        GenerationTimeMs: int(generationTime),
    }
    
    if err := h.db.SaveNarrative(dbNarrative); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "failed to save narrative: " + err.Error()})
        return
    }
    
    c.JSON(http.StatusOK, gin.H{
        "narrative": dbNarrative,
        "cached":    false,
    })
}

func (h *NarrativeHandler) GetNarrative(c *gin.Context) {
    incidentID, err := strconv.ParseInt(c.Param("id"), 10, 64)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "invalid incident id"})
        return
    }
    
    narrative, err := h.db.GetNarrativeByIncident(incidentID)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }
    if narrative == nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "narrative not found"})
        return
    }
    
    c.JSON(http.StatusOK, gin.H{"narrative": narrative})
}

func (h *NarrativeHandler) GetNarrativeDetail(c *gin.Context) {
    narrativeID, err := strconv.ParseInt(c.Param("id"), 10, 64)
    if err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "invalid narrative id"})
        return
    }
    
    narr, err := h.db.GetNarrativeByID(narrativeID)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }
    if narr == nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "narrative not found"})
        return
    }
    
    // Parse sentences to extract source event IDs
    var sentences []narrative.NarrativeSentence
    if err := json.Unmarshal(narr.Sentences, &sentences); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "failed to parse sentences"})
        return
    }
    
    // Collect all source event IDs
    eventIDMap := make(map[int64]bool)
    for _, s := range sentences {
        for _, id := range s.SourceEventIDs {
            eventIDMap[id] = true
        }
    }
    
    // Fetch source events
    sourceEvents := make(map[int64]interface{})
    for eventID := range eventIDMap {
        // Query events table directly
        var event database.Event
        err := h.db.Conn().QueryRow(`
            SELECT id, timestamp, hostname, event_type, source_ip, raw_json
            FROM events WHERE id = ?
        `, eventID).Scan(&event.ID, &event.Timestamp, &event.Hostname, &event.EventType, &event.SourceIP, &event.RawJSON)
        if err == nil {
            sourceEvents[eventID] = event
        }
    }
    
    c.JSON(http.StatusOK, gin.H{
        "narrative":     narr,
        "source_events": sourceEvents,
    })
}
```
